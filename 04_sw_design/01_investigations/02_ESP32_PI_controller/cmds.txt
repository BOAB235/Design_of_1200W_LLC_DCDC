# remove build folders 
Get-ChildItem -Directory | ForEach-Object {
    $buildPath = Join-Path $_.FullName "build"
    if (Test-Path $buildPath) {
        Write-Host "Removing build folder in $($_.Name) ..."
        Remove-Item $buildPath -Recurse -Force
    }
}
